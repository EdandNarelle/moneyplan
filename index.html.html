function doPost(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var data = JSON.parse(e.postData.contents); if (data.action === 'delete') { var delSheet = ss.getSheetByName(data.sheetName); if (delSheet) { var delRows = delSheet.getDataRange().getValues(); for (var i = delRows.length - 1; i >= 1; i--) { if (String(delRows[i][5]) === String(data.ts)) { delSheet.deleteRow(i + 1); break; } } } return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } var monthName = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "MMM yyyy"); var sheet = ss.getSheetByName(monthName); if (!sheet) { sheet = ss.insertSheet(monthName); sheet.appendRow(["Date","Shop","Amount","Category","Notes","ID"]); var hdr = sheet.getRange(1,1,1,6); hdr.setFontWeight("bold"); hdr.setBackground("#1a7a4a"); hdr.setFontColor("white"); sheet.setColumnWidth(1,90); sheet.setColumnWidth(2,150); sheet.setColumnWidth(3,80); sheet.setColumnWidth(4,150); sheet.setColumnWidth(5,200); sheet.setColumnWidth(6,100); } var today = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "dd/MM/yyyy"); sheet.appendRow([today, data.shop, data.amount, data.category, data.notes||"", data.ts]); updateDashboard(ss); return ContentService.createTextOutput(JSON.stringify({ok:true})).setMimeType(ContentService.MimeType.JSON); } function doGet(e) { var ss = SpreadsheetApp.getActiveSpreadsheet(); var monthName = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "MMM yyyy"); if (e.parameter.month) monthName = e.parameter.month; var sheet = ss.getSheetByName(monthName); if (!sheet) return ContentService.createTextOutput(JSON.stringify({rows:[]})).setMimeType(ContentService.MimeType.JSON); var rows = sheet.getDataRange().getValues(); var result = []; for (var i = 1; i < rows.length; i++) { if (rows[i][0]) result.push({date:String(rows[i][0]),shop:rows[i][1],amount:rows[i][2],category:rows[i][3],notes:rows[i][4],ts:rows[i][5]}); } return ContentService.createTextOutput(JSON.stringify({rows:result,month:monthName})).setMimeType(ContentService.MimeType.JSON); } function updateDashboard(ss) { var dash = ss.getSheetByName("Dashboard"); if (!dash) { dash = ss.insertSheet("Dashboard", 0); dash.getRange("A1").setValue("Ed & Narelle — Money Plan Dashboard"); dash.getRange("A1").setFontSize(16).setFontWeight("bold").setFontColor("#1a7a4a"); dash.getRange("A3").setValue("BANK & ASSETS"); dash.getRange("A3").setFontWeight("bold"); dash.getRange("A4:B4").setValues([["Current Account",""]]); dash.getRange("A5:B5").setValues([["Savings Account",""]]); dash.getRange("A6:B6").setValues([["House Value",""]]); dash.getRange("A7:B7").setValues([["Mortgage Balance",""]]); dash.getRange("A8:B8").setValues([["Pensions",""]]); dash.getRange("A9:B9").setValues([["Other Assets",""]]); dash.getRange("A11").setValue("THIS MONTH SPENDING"); dash.getRange("A11").setFontWeight("bold"); dash.setColumnWidth(1,200); dash.setColumnWidth(2,150); dash.setColumnWidth(3,150); } var monthName = Utilities.formatDate(new Date(), Session.getScriptTimeZone(), "MMM yyyy"); var sheet = ss.getSheetByName(monthName); if (!sheet) return; var rows = sheet.getDataRange().getValues(); var cats = {}; var total = 0; for (var i = 1; i < rows.length; i++) { if (!rows[i][0]) continue;
